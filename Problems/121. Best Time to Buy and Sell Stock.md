class Solution(object):
    def maxProfit(self, prices):
        """
        :type prices: List[int]
        :rtype: int
        """
        profit = {}
        left = prices[0]
        i = 0

        for n in prices:
            if n < left:
                left = n
            else:
                profit[n] = n - left
            i += 1
        if left == prices[i-1]:
            return 0
        else:
            return max(profit.values())
            # return profit[max(profit, key = profit.get)]
