class Solution(object):
    def maxProfit(self, prices):
        """
        :type prices: List[int]
        :rtype: int
        """
        profit = {}
        left, right = 0, 1
        sums=0

        while right <  len(prices):
            profit[prices[left]] = prices[right] - prices[left]
            right += 1
            left += 1
        positives = [value for value in profit.values() if value > 0]
        sums = sum(positives)
        
        return sums
