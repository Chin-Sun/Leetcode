## Better Solution--- HashMap + for 
```
class Solution(object):
    def romanToInt(self, s):
        """
        :type s: str
        :rtype: int
        """

        roman = {"I": 1, "V": 5, "X": 10, "L": 50, "C": 100, 
        "D":500,"M":1000}

        sums = 0
        for i in range(len(s)):
            if i+1<len(s) and roman[s[i]] < roman[s[i+1]]:
                sums -= roman[s[i]]
            else:
                sums += roman[s[i]]
        return sums
```

    def romanToInt(self, s):
        """
        :type s: str
        :rtype: int
        """
        sums = 0

        for i in range(len(s)):
            if s[i] == "M":
                sums += 1000
            elif s[i] == "D":
                sums += 500
            elif s[i] == "L":
                sums += 50
            elif s[i] == "V":
                sums += 5
            elif s[i] == "I":
                if i+1 <len(s) and (s[i+1] == "V" or s[i+1] == "X") :
                    sums -= 1
                else: 
                    sums += 1
            elif s[i] == "X":
                if i+1 <len(s) and (s[i+1] == "L" or s[i+1] == "C" ):
                    sums -= 10
                else: 
                    sums += 10
            elif s[i] == "C":
                if i+1 <len(s) and (s[i+1] == "D" or s[i+1]=="M" ) :
                    sums -= 100
                else: 
                    sums += 100
        return sums
